 12 /**
 13  * 	@namespace Holds log level constants for the framework. Set 
 14  * {@link IMAT.settings.logLevel} to one of the Constants to throttle the 
 15  * logging threashold.
 16  */
 17 IMAT.logLevels = {
 18 	/**
 19 	 * When {@link IMAT.settings.logLevel} is set to LOG_OFF, nothing is logged.
 20 	 * @constant
 21 	 * @type number 
 22 	 */
 23 	LOG_OFF: 0,
 24 	
 25 	/**
 26 	 * When {@link IMAT.settings.logLevel} is set to LOG_ERROR, only errors are
 27 	 * logged. 
 28 	 * @constant 
 29 	 * @type number 
 30 	 */
 31 	LOG_ERROR: 1,
 32 	
 33 	/**
 34 	 * When {@link IMAT.settings.logLevel} is set to LOG_WARN, only errors and
 35 	 * warning messages are logged. 
 36 	 * @constant
 37 	 * @type number 
 38 	 */
 39 	LOG_WARN: 2,
 40 	
 41 	/**
 42 	 * When {@link IMAT.settings.logLevel} is set to LOG_INFO, only errors, 
 43 	 * warnings, and info messages are logged. This is good for running your 
 44 	 * tests in an unattended state, because it provides a good level of 
 45 	 * granularoty in order to adequately triage your tests when things go 
 46 	 * wrong.
 47 	 * @constant
 48 	 * @type number 
 49 	 */
 50 	LOG_INFO: 3,
 51 	
 52 	/**
 53 	 * When {@link IMAT.settings.logLevel} is set to LOG_DEBUG, only errors, 
 54 	 * warnings, info, and debug messages are logged. This is good for 
 55 	 * developing your tests.
 56 	 * @constant
 57 	 * @type number 
 58 	 */
 59 	LOG_DEBUG: 4,
 60 	
 61 	/**
 62 	 * When {@link IMAT.settings.logLevel} is set to LOG_TRACE, log everything.
 63 	 * It is recommended you ony use this when you are running into problems
 64 	 * that you cannot track down.  
 65 	 * @constant
 66 	 * @type number 
 67 	 */
 68 	LOG_TRACE: 5
 69 };
