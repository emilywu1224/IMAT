 12 /**
 13  * 	Constructs a SuiteRunner object
 14  *	@class The {@link IMAT.SuiteRunner} object is the highest level object 
 15  *  managing the test execution lifecycle. It's major responsibilities include:
 16  *  <ul>
 17  *		<li>Providing an API for suite files of an automation project to run and
 18  *  	manage tests</li>
 19  *		<li>Giving a preview of all tests considered runnable given it's current 
 20  * 		state</li>
 21  *  	<li>Maintaining an array of Test Sets.</li>
 22  *  </ul>
 23  * <b>IMPORTANT NOTE</b>: This class wraps the {@link IMAT.TestRunner} object, 
 24  * so you should never create one yourself or call it directly.
 25  * @memberOf IMAT 
 26  */
 27 IMAT.SuiteRunner = Class.create(/** @lends IMAT.SuiteRunner# */{
 28 	
 29 	/**
 30 	 * The constructor called when invoking "new" on this class object.
 31 	 * @ignore 
 32 	 */
 33 	initialize: function()
 34 	{
 35 		this.testSetArray = [];
 36 		this.testReporter = new IMAT.TestReporter();
 37 		var testRunner = new IMAT.TestRunner();
 38 		testRunner.setTestReporter(this.testReporter);
 39 		this.setTestRunner(testRunner);
 40 	},
 41 	
 42 	/**
 43 	 * Runs a given test case
 44 	 * 
 45 	 * @param {SuiteHander} suiteHandler 
 46 	 * 					An object that implements the following functions:
 47 	 * <ul>
 48 	 * 		<li><code>initSuite()</code></li>
 49 	 *		<li><code>cleanUpSuite()</code></li>
 50 	 * </ul>
 51 	 */
 52 	runTests: function(suiteHandler)
 53 	{
 54 		var testSet;
 55 		IMAT.log_debug("Running Suite...");
 56 		if(suiteHandler)
 57 		{
 58 			var initSuiteToken = "suiteHandler.initSuite();"; 
 59 			var cleanUpSuiteToken = "suiteHandler.cleanUpSuite();";
 60 			IMAT.log_start(initSuiteToken);
 61 			if (!IMAT.settings.SKIP_REPORT_PROJECTION) {
 62 				this.reportOnProjectedTests();
 63 			}
 64 			if (!IMAT.settings.SKIP_INIT_SUITE) {
 65 				suiteHandler.initSuite();
 66 			}
 67 			IMAT.log_pass(initSuiteToken);
 68 			testSet = this.testSetArray.pop();
 69 			while (testSet)
 70 			{
 71 				this.testRunner.runTestCases(testSet);
 72 				testSet = this.testSetArray.pop();
 73 			}
 74 			IMAT.log_start(cleanUpSuiteToken);
 75 			if (!IMAT.settings.SKIP_CLEAN_UP_SUITE) {
 76 				suiteHandler.cleanUpSuite();
 77 			}
 78 			if (!IMAT.settings.SKIP_REPORT_RETROSPECTIVE) {
 79 				this.testReporter.reportRetrospective();
 80 			}
 81 			IMAT.log_pass(cleanUpSuiteToken);
 82 		}
 83 		else
 84 		{
 85 			throw "suiteHandler was not defined when calling SuiteRunner.runTests(...)";	
 86 		}
 87 	},
 88 	
 89 	reportOnProjectedTests: function() {
 90 		var tests = this.collectAllRunnableTests();
 91 		for (var i = 0; i < tests.length; i++) {
 92 			this.testReporter.addTest(tests[i]);
 93 		}
 94 		this.testReporter.reportProjection();
 95 	},
 96 	
 97 	
 98 	collectAllRunnableTests: function()
 99 	{
100 		var runnableTests = [];
101 		for(var i = 0; i < this.testSetArray.length; i++)
102 		{
103 			this.testRunner.collectRunnableTests(this.testSetArray[i], 
104 				runnableTests);
105 		}
106 		return runnableTests;
107 	},
108 	
109 	/**
110 	 * Log a list of testCases that will be run when 
111 	 * {@link IMAT.SuiteRunner.runTests} is called. Call this function after 
112 	 * setting the filters. <b>NOTE:</b> Since this should be used to debug 
113 	 * which tests are included in a filter, this should be run when 
114 	 * {@link IMAT.settings.logLevel} is set to {@link IMAT.logLevels.LOG_DEBUG}
115 	 * or higher. 
116 	 *
117 	 *  @example
118 	 *  //What follows is a snippet of a suite file located in the suites/ 
119 	 *  //directory of an automation project generated by IMAT. If run, it would
120 	 *  //spit out a log with all of the tests matching the filters that were set
121 	 *
122 	 *  //run the tests
123 	 *  IMAT.suiteRunner.addFilters(["List.*\\.test.*Search"]);
124 	 *  IMAT.suiteRunner.previewAllRunnableTests();
125 	 */
126 	previewAllRunnableTests: function()
127 	{
128 		IMAT.log_debug("Previewing all runnable tests...");
129 		for(var i = 0; i < this.testSetArray.length; i++)
130 		{
131 			this.testRunner.previewRunnableTests(this.testSetArray[i]);
132 		}
133 	},
134 	
135 	/**
136 	 * Adds a set of test cases to the current list of test cases to run. By 
137 	 * convetion we normally call this after declaring a test set. See the 
138 	 * example below. 
139 	 *
140 	 * @param {IMAT.BaseTestSet} testSet 
141 	 * 					The object containing a set of test objects to run.
142 	 * @example
143 	 * EXAMPLE.MyTestSet = Class.extend(IMAT.BaseFunctionalTest, {
144 	 * 		...Tests would normally go here...
145 	 * });
146 	 *  
147 	 * IMAT.suiteRunner.addTestSet( new EXAMPLE.MyTestSet()); 
148 	 */
149 	addTestSet: function(testSet)
150 	{		
151 		this.testSetArray.push(testSet);
152 	},
153 	
154 	/**
155 	 * Add filters to the current set maintained by the TestRunner object. All 
156 	 * filters added will be concatenated with the existing set. <b>Note:</b>
157 	 * this function can be called more than once if you have the need to split 
158 	 * your test filtering accross mulitple places. 
159 	 *
160 	 * @param {array} filterArray 
161 	 * 						An array of strings representing filters. NOTE  
162 	 * 						that strings given representing regular expressions 
163 	 * 						are supported and encouraged! 
164 	 * 
165 	 * Filters represented as Regular expressions do not need to include the 
166 	 * wrapping "/" characters. Also note that modifiers like g, i etc are not 
167 	 * supported. Since the regular expressions are represented as strings, all 
168 	 * "\" should be escaped like "\\" 
169 	 *
170 	 * @example 
171 	 * //Selects all test cases with the word "search" in the signature.
172 	 * IMAT.suiteRunner.addFilters(["search"]);
173 	 * 
174 	 * //Selects all test cases with the word "search" or the word "navigate" in
175 	 * //the signature.
176 	 * IMAT.suiteRunner.addFilters(["search", "navigate"]); 
177 	 *
178 	 * //Selects all test cases whose Test Sets have the word "List" in their
179 	 * //title and whose method signatures have the word "Search"
180 	 * //Also note the double backslash we used to escape the period part of the
181 	 * //total test signature.
182 	 * IMAT.suiteRunner.addFilters(["List.*\\.test.*Search"]); 
183 	 * 
184 	 * //would give...
185 	 * DEBUG: 	TestSet: ItemListTestSet
186 	 * DEBUG: 		ItemListTestSet.testSearchForNonExistantItems()
187 	 */
188 	addFilters: function(filterArray)
189 	{		
190 		this.testRunner.addFilters(filterArray);
191 	},
192 	
193 	/**
194 	 *  Set the testRunner
195 	 *
196 	 *  @param {IMAT.TestRunner} runner 
197 	 * 					The test runner object we wish to set. 
198 	 * 
199 	 *  @ignore 
200 	 */
201 	setTestRunner: function(runner)
202 	{
203 		this.testRunner = runner;
204 	}
205 	
206 });
207 // Declare a global suite runner object that we can use to run our tests.
208 IMAT.suiteRunner = new IMAT.SuiteRunner();
