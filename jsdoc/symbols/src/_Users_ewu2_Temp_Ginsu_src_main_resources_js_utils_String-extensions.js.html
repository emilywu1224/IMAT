 12 /**
 13  * Determine if this string object starts with the given string. This is an extension to the base
 14  * String object of the JavaScript language.
 15  * 
 16  * @param {String} str 
 17  * 					The string to compare this against
 18  * 
 19  * @return {boolean} True if this string object starts with str, false otherwise.
 20  * @memberOf String
 21  */
 22 String.prototype.startsWith = function(str)
 23 {
 24 	return (this.match( "^" + str ) == str);
 25 };
