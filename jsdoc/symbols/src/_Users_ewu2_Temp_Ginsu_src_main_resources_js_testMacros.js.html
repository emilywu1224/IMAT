 12 /**
 13  * Checks for Strict equality. Depending on if you are using objects or 
 14  * primitives, this will work differently. JS is pass by reference for objects, 
 15  * so if you are testing that two objects are equal, this will only work if they 
 16  * are the same reference. For literals, this is simply checking that they 
 17  * evaluate to the same value.
 18  *
 19  * @param expected The expected value we are testing
 20  * @param received The value we expected to see
 21  * @param message (optional) The message to throw if expected did not match 
 22  * received
 23  *
 24  * @throws An exception string with the given parameter message if expected was 
 25  * not strictly equal to received.
 26  */
 27  /*
 28  * TODO: Create an assertEqualObjects function that calls isEqual on the 
 29  * expected object. Consider underscore.js
 30  */
 31 function assertEquals(received, expected, message) {
 32   if (received !== expected) {
 33     if (!message) {
 34     	message = "Expected " + expected + " (type: " + typeof expected + ")" + 
 35     		" but received " + received + " (type: " + typeof received + 
 36     		"). Check for value and type.";
 37     }
 38     throw new IMAT.AssertionException(message);
 39   }
 40 }

 42 /**
 43  * Checks for Strict inequality. Depending on if you are using objects or 
 44  * primitives, this will work differently. JS is pass by reference for objects, 
 45  * so if you are testing that two objects are not equal, this will throw an 
 46  * exception if they are the same reference. For literals, this is strictly 
 47  * checking to make sure they are different values.
 48  *
 49  * @param expected The expected value we are testing
 50  * @param received The value we expected to see
 51  * @param message (optional) The message to throw if expected matched received
 52  *
 53  * @throws An exception string with the given parameter message if expected  
 54  * was strictly equal to received.
 55  */
 56 function assertNotEquals(received, expected, message) {
 57   if (received === expected) {
 58     if (!message) {
 59     	message = "Expected " + expected + " (type: " + typeof expected + ")" + 
 60     		" but received " + received + " (type: " + typeof received + 
 61     		"). Check for value and type.";
 62     }
 63     throw new IMAT.AssertionException(message);
 64   }
 65 }

 69 /**
 70  * Checks that the expression evaluates to true.
 71  *
 72  * @param expression The expression value we are checking
 73  * @param message (optional) The message to throw if expression does not 
 74  * evaluate to true
 75  *
 76  * @throws an error message if expression does not evaluate to true.
 77  */
 78 function assertTrue(expression, message) {
 79   if (!expression) {
 80     if (!message){
 81     	message = "Assertion failed when expression was: " + expression + 
 82     		" type: " + typeof expression;
 83     } 
 84     throw new IMAT.AssertionException(message);
 85   }
 86 }
 87 
 88 /**
 89  * Checks that the expression evaluates to false.
 90  *
 91  * @param expression The expression value we are checking
 92  * @param message (optional) The message to throw if expression does not 
 93  * evaluate to false
 94  *
 95  * @throws an error message if expression does not evaluate to false.
 96  */
 97 function assertFalse(expression, message) {
 98   assertTrue(!expression, message);
 99 }

101 /**
102  * Checks that thingie does not evaluate to null or any of its equivalents 
103  * (i.e. undefined). Note that this also handles cases where UIAElementNil is 
104  * returned. However, it does not consider the value false to be an equivalent 
105  * of null. For that use the assertFalse() function.
106  *
107  * @param thingie The expected value we are testing
108  * @param message (optional) The message to throw if thingie evaluated to a null 
109  * equivalents
110  *
111  * @throws an error message if expected did not match received
112  */
113 function assertNotNull(thingie, message) {
114   if (thingie === null || typeof thingie === "undefined" || thingie instanceof UIAElementNil) {
115     if (!message) {
116     	message = "Expected a non null object. Object was " + thingie;
117     }
118     throw new IMAT.AssertionException(message);
119   }
120 }

122 /**
123  * Checks that two values from a sort are in an ascending relationship.
124  *
125  * @param lowValue The value that should be the lower of the two values.
126  * @param highValue The value that should be the higher of the two values.
127  *
128  * @throws an error if the values are not in ascending order.
129  */
130 function assertAscending(lowValue, highValue, msg)
131 {
132 	if (lowValue >= highValue)
133 	{
134 		var message = msg + " Sorted values are not in ascending order. highValue " + 
135 			highValue + " is not higher than lowValue " + lowValue; 
136 		throw new IMAT.AssertionException(message);
137 	}
138 }

140 /**
141  * Checks that two values from a sort are in an descending relationship.
142  *
143  * @param highValue The value that should be the higher of the two values.
144  * @param lowValue The value that should be the lower of the two values.
145  *
146  * @throws an error if the values are not in descending order.
147  */
148 function assertDescending(highValue, lowValue, msg)
149 {
150 	if (lowValue >= highValue)
151 	{
152 		var message = msg + " Sorted values are not in descending order. lowValue " + 
153 			lowValue + " is not lower than highValue " + highValue;
154 		throw new IMAT.AssertionException(message);
155 	}
156 }

158 /**
159  * For keeping track of manual tests that have not been automated yet, this will
160  * log a message stating that a given test has passed manually. This is meant to
161  * be called directly within a test.
162  *
163  *  @param {string} message
164  * 					An optional message to log with the manual pass message. 
165  */
166 function manualPass()
167 {
168 	var message = "MANUAL: This test is marked as Passing during manual execution.";
169 	if (arguments.length > 0) {
170 		message = message + " " + arguments[0];
171 	} 
172 	IMAT.log_info(message);
173 }
174 
175 /**
176  * For keeping track of manual tests that have not been automated yet, this will
177  * log a message stating that a given test has failed manually. This is meant to
178  * be called directly within a test.
179  *
180  *  @param {string} message
181  * 					An optional message to log with the manual fail message. 
182  */
183 function manualFail()
184 {
185 	var message = "MANUAL: This test is marked as failing during manual execution.";
186 	if (arguments.length > 0) {
187 		message = message + " " + arguments[0]; 
188 	} 
189 	throw new IMAT.AssertionException(message);
190 }
