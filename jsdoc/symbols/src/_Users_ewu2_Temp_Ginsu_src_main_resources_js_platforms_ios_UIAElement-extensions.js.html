 12 /**
 13  * Wait for a given element to be invisible. Uses the local target's timeout 
 14  * value and checks for isVisible until the timeout time is reached. This is an 
 15  * extension to the UIAElement object of the UI Automation framework.
 16  * 
 17  * @param {boolean} softTimeout 
 18  * 					Optional. Set to true if we should log the timeout at the  
 19  * 					debug level rather than as a warning.
 20  * 
 21  * @memberOf UIAElement#
 22  */
 23 UIAElement.prototype.waitForInvisible = function(softTimeout)
 24 {
 25 	//Since it is possible for the timeout to be verrrry long, 30 seconds or 
 26 	//more, we need to store its value and replace the top timeout with a 1 
 27 	//second timeout value.
 28 	var timeout = UIATarget.localTarget().timeout();
 29 	UIATarget.localTarget().pushTimeout(1);
 30 	
 31 	var endDate, startDate;
 32 	startDate = endDate = new Date();
 33 	var millisecondsPerSecond = 1000;
 34 	var iAmVisible = true;
 35 	IMAT.log_trace("waitForInvisible: " + this);
 36 	
 37 	//Internal convenience function to get the time since we started waiting for 
 38 	//the element to be invisible.
 39 	var timeSinceWaitStart = function()
 40 	{
 41 		return (endDate.getTime() - startDate.getTime())/millisecondsPerSecond;
 42 	};
 43 	
 44 	while (iAmVisible && timeSinceWaitStart() < timeout)
 45 	{
 46 		iAmVisible = this && this.checkIsValid() && this.isVisible();
 47 		endDate = new Date();
 48 		if(timeSinceWaitStart() >= timeout)
 49 		{
 50 			var message = "Timeout Exceeded: waitForInvisible | timeout: " + timeout + " sec | " + this;
 51 			if(softTimeout)
 52 			{
 53 				IMAT.log_debug(message);
 54 			}
 55 			else
 56 			{
 57 				IMAT.log_warning(message);
 58 			}
 59 			
 60 		}
 61 	}
 62 	//Reset the timeout stack back to its original state once we are done waiting.
 63 	UIATarget.localTarget().popTimeout();
 64 };
