 12 /**
 13  * 	Constructs a BaseFunctionalTestSet object
 14  *	@class The {@link IMAT.BaseFunctionalTestSet} object is common test set 
 15  * 	object to be extended by all tests doing testing at the UI level. 
 16  * 	<p>Since this
 17  *  framework is primarily intended to be a UI testing framework, we assume that
 18  *  most test sets will extend this class. However, if those assumptions prove
 19  *  incorrect and you find yourself needing something entirely different, you 
 20  *  are free to omit this as a parent class to your Test Sets and implement 
 21  *  something that makes sense for  you.</p>
 22  *  <p>On the other hand, if you are doing UI testing, it is strongly recommended
 23  *  that you take advantage of the implementations this class provides.</p> 
 24  *  <ul>
 25  *		<li>Alert management at the test level so that you can test for expected
 26  * 		alerts and fail out when certain types of unexpected alerts occur.</li>
 27  *		<li>Clean-up hooks to reset your application to a know good state after
 28  * 		an exception or a failure has occured.</li>
 29  *  	<li>The Perform actions function. This is the ""miracle method" that 
 30  * 		allows our tests to be readable as well as easy to triage.</li>
 31  *  </ul>
 32  * @memberOf IMAT 
 33  * @augments IMAT.BaseTestSet 
 34  */

 48 	/**
 49 	 * The alert manager used for testing scenarios when alerts are expected and 
 50 	 * the user is suposed to take some sort of action.
 51 	 *
 52 	 * @example
 53 	 * //...within some Test Set...
 54 	 * testIncorrectPasswordOnLogin: function() {
 55 	 * 	this.alertManager.handleAlertWithTitleAndButtonClick("Incorrect Password", "Reset");
 56 	 *	this.performActions([
 57 	 * 		["login", "testUser", "badPassword"],
 58 	 * 		["wait", 10], //wait for the alert to pop up and be handled.
 59 	 * 		["validateResetPassword"]
 60 	 * 	]); 
 61 	 * 	this.alertManager.resetAlertHandler();
 62 	 * },
 63 	 */
 64 	alertManager : /** @lends IMAT.BaseFunctionalTestSet.alertManager */{
 65 		
 66 		defaultAlertHandler : UIATarget.onAlert,
 67 	
 68 		/**
 69 		 * Sets the alert handler to handle only alerts with the given title. 
 70 		 * The alert handler will return true if the title matches the alert 
 71 		 * triggered, however calling this function explicitly will not return 
 72 		 * anything.
 73 		 * 
 74 		 * @param {String} title
 75 		 *  					The title of the UIAAlert to handler
 76 		 * @throws {String} when the value of alert.name() does not equal title. 
 77 		 */
 78 		handleDefaultAlertWithTitle : function(title) {
 79 			// need an internal reference to pass to the callback method ('this' changes context)
 80 			var ref = this;
 81 			UIATarget.onAlert = function onAlert(alert) {
 82 				ref.validateAlertWithTitle(alert, title);
 83 				return true;
 84 			};
 85 		},
 86 		
 87 		validateAlertWithTitle : function(alert, title) {
 88 			IMAT.log_debug("Validating alert named: '" + title + "'");			
 89 			if (IMAT.settings.logLevel >= IMAT.logLevels.LOG_DEBUG) {
 90 				alert.logElementTree();
 91 			}
 92 			// if the name field exists then check that
 93 			var alertTitle = null;
 94 			if(alert.name()) {
 95 				alertTitle = alert.name();
 96 			// if not then look for a text field with the title we are looking for
 97 			} else {
 98 				alertTitle = alert.elements().firstWithName(title).name();
 99 			}
100 			assertEquals(alertTitle, title);
101 		},
102 	
103 		/**
104 		 * 
105 		 */
106 		handleAlertWithTitleAndButtonClick : function(title, buttonName) {
107 			// need an internal reference to pass to the callback method ('this' changes context)
108 			var ref = this;
109 			UIATarget.onAlert = function onAlert(alert) {
110 				ref.validateAlertWithTitle(alert, title);
111 				IMAT.log_debug("Tapping Alert Button: '" + buttonName + "'");
112 				// tap tap taparoo that button with the given name
113 				alert.buttons().firstWithName(buttonName).tap();
114 				return true;
115 			};
116 		},
117 		
118 		/**
119 		 * Set the handler back to the default one after we are done.
120 		 */
121 		resetAlertHandler : function() {
122 			UIATarget.onAlert = this.defaultAlertHandler;
123 		},
124 		
125 		/**
126 		 * Set the handler back to the default one after we are done. Pass the 
127 		 * contextView
128 		 */
129 		resetAlertHandler : function(viewContext) {
130 			UIATarget.onAlert = this.defaultAlertHandler;
131 			//wait for the next view coming up after the alert is handled.
132 			viewContext.target.delay(10);
133 		}
134 	},

164 	/**
165 	 * Do any cleanup needed to make sure that subsequent tests run in a known 
166 	 * good state. This function iteratively calls escapeAction on the current 
167 	 * view context until the returned context is not set.
168 	 *
169 	 * If the escapeAction ever throws an exception, we attempt to discover what 
170 	 * view we are on and try the escape process again. This will happen up to 5 
171 	 * times before we give up.
172 	 *
173 	 *  @param {number} attemptCount 
174 	 * 					The number of attempt counts we have left.
175 	 */
176 	doCleanup: function(attemptCount) { 
177 		var stemView;
178 		//initialize the attempt
179 		if (typeof attemptCount == "undefined")
180 		{
181 			attemptCount = 0;
182 		}
183 		//attempt the whole escape process up to 5 times
184 		if (attemptCount < 5)  {
185 			try {
186 				do {
187 					IMAT.log_debug("Escaping from view: " + this.viewContext.viewName);
188 					if (this.viewContext.viewName != this.getBaseStateViewName()) {
189 						this.viewContext = this.viewContext.escapeAction();
190 					}
191 				}
192 				while(this.viewContext && this.viewContext.viewName != this.getBaseStateViewName());
193 				
194 				//we should now be at the login view. If we are not, trying to instantiate one here
195 				// will throw an exception and cause us to try again.
196 				this.viewContext = this.getBaseView();
197 			} catch (e) {
198 				IMAT.log_debug("Escape failed when attempt count was " + attemptCount);
199 				stemView = new IMAT.StemView();
200 				this.viewContext = stemView.getSpecializedView();
201 				//make a recursive call to doCleanup and increment the attemptCount
202 				this.doCleanup(++attemptCount);
203 			}
204 		} else {
205 			IMAT.log_error("Automation failed cleanup more than 5 times in a row.");
206 		}
207 	},

209 	/**
210 	 * Get the name of the view which we consider to be the "Base State" when 
211 	 * recovering from a test failure.This implementation does nothing and 
212 	 * returns undefined. You must override this in order to get test recovery 
213 	 * working correctly. 
214 	 *
215 	 * @return String The name of our base view for all functional tests.
216 	 */
217 	getBaseStateViewName: function() {
218 		return	undefined;
219 	},

223 	 * recovering from a test failure. This implementation does nothing and 
224 	 * returns undefined. You must override this in order to get test recovery 
225 	 * working correctly. 
226 	 *
227 	 * @return {IMAT.BaseView} The View we are trying to get to.
228 	 */
229 	getBaseView: function() {
230 		return undefined;
231 	},

233 	/**
234 	 * Performs a set of actions defined by the two dimensional actions array 
235 	 * passed in.
236 	 * <p>When perform actions is called, it automatically updates the view
237 	 * context to the resulting view of each action. Furthermore it logs action
238 	 * information to aid in triage when tests fail.</p> 
239 	 * <p>Notice the example below. Each action array starts with the name of 
240 	 * the action, and is followed by the parameters to that action. This is 
241 	 * also a very easy format to read what the test does and understand it
242 	 * without much prior knowledge of JavaScript or the framework within.</p>
243 	 *  
244 	 * @example
245 	 * //...within some test set....
246 	 * testSearchEnterTermAndClear: function() {
247 	 * 	this.performActions([
248 	 * 		["search", "Carrots"],
249 	 *		["validateItemPresent", "Carrots"],
250 	 *		["clearSerachTerm"]
251 	 * 	]);
252 	 * },
253 	 *
254 	 *  @param {Array} actions
255 	 * 					Where each element in the actions array is another 
256 	 * 					{@link Array}. The first element in the sub-array should 
257 	 *  				be a string representing an action method of the current
258 	 *  				viewContext (<code>this.viewContext</code>) of the test. 
259 	 * 					The rest of the elements in the array are interpreted as
260 	 * 					parameters to that method.
261 	 */
262 	performActions: function(actions) {
263 		var typeCheckMessage = "Perform actions must take a parameter of a 2 dimensional array.";
264 		assertTrue((actions instanceof Array), typeCheckMessage);
265 		var i;
266 		for (i = 0; i < actions.length; i++) {
267 			assertTrue((actions[i] instanceof Array), typeCheckMessage);
268 			assertTrue(actions[i].length > 0, "All actions require at least one string representing the action.");
269 			var action = actions[i][0] + "Action";
270 			var logMessage = "ACTION: " + actions[i][0] + "("; 
271 			var j; 
272 			var args = [];
273 			for (j = 1; j < actions[i].length; j++)
274 			{
275 				args.push(actions[i][j]);
276 				logMessage = logMessage + " " + actions[i][j];
277 			}
278 			logMessage = logMessage + ")";
279 			IMAT.log_info(logMessage);
280 			var actionFunction = this.viewContext[action];
281 			assertTrue((typeof actionFunction == "function"),
282 				action + " is not an action of " + this.viewContext.viewName +
283 				" instead it evaluates to: " + actionFunction);
284 			this.viewContext = actionFunction.apply(this.viewContext, args);
285 		}
286 	}
