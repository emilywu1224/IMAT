340 	testManualFailMacro: function() {
341 		var latestException = undefined;
342 		IMAT.log_debug("Testing with no arguments");
343 		latestException = this.manualfailHelper(function(){manualFail();});
344 		
345 		var expectedMessage = "With message";
346 		IMAT.log_debug("Testing with 1 argument");
347 		latestException = this.manualfailHelper(function(){manualFail(expectedMessage);});
348 		assertTrue((latestException.message.indexOf(expectedMessage) > -1), "The " +
349 			"exception did not contain the expected string. Exception: " +
350 			latestException);
351 		
352 		//2nd param will not show up in logs
353 		var unexpectedMessage  ="And Another";
354 		IMAT.log_debug("Testing with 2 arguments");
355 		latestException = this.manualfailHelper(function(){manualFail(expectedMessage, unexpectedMessage);});
356 		assertTrue((latestException.message.indexOf(expectedMessage) > -1), "The " +
357 			"exception did not contain the expected string. Exception: " +
358 			latestException);
359 		assertTrue((latestException.message.indexOf(unexpectedMessage) == -1), "The " +
360 			"exception had the unexpected string. Exception: " +
361 			latestException);
362 	},

364 	manualfailHelper: function(func) {
365 		var didThrowException = true;
366 		var result = undefined;
367 		try {
368 			func();
369 			didThrowException = false; // we should never get here.
370 		} catch (e){
371 			assertTrue(e instanceof IMAT.AssertionException, "the exception thrown should have been of type IMAT.AssertionException");
372 			assertTrue((e.message.indexOf("MANUAL:") > -1), "The exception did not contain the expected string.");
373 			result = e;
374 		}
375 		assertTrue(didThrowException, 
376 			"manualFail() did not throw an exception when it should have");
377 		return result;
378 	},

380 	testReportingAddTest: function() {
381 		var reporter = new IMAT.TestReporter();
382 		assertTrue(reporter.automatedTests.length === 0, "An empty test reporter should have no automated tests");
383 		assertTrue(reporter.manualTests.length === 0, "An empty test reporter should have no manual tests");
384 		
385 		//add a manual test
386 		var manualSignature = "testSomething.testManuallyFoobar";
387 		reporter.addTest(manualSignature);
388 		assertEquals(reporter.automatedTests.length, 0, "A reporter with one manual test should have 0 automated tests");
389 		assertEquals(reporter.manualTests.length, 1);
390 		assertFalse(reporter.manualTests[manualSignature].hasRun, "has run should be false");
391 		assertEquals(typeof reporter.manualTests[manualSignature].result, "undefined");
392 		
393 		//Add an automated test
394 		var automatedSignature = "testSomething.testFoobar";
395 		reporter.addTest(automatedSignature);
396 		assertTrue(reporter.automatedTests.length == 1, "After adding automated test should have 1 automated tests");
397 		assertTrue(reporter.manualTests.length == 1, "After adding automated test should still have 1 manual test");
398 		assertFalse(reporter.automatedTests[automatedSignature].hasRun, "has run should be false");
399 		assertEquals(typeof reporter.automatedTests[automatedSignature].result, "undefined");
400 	},

452 		var manualSignatures = ["testSomething.testManuallyPass", 
453 			"testSomething.testManuallyError", 
454 			"testSomething.testManuallyFailure"];

462 			IMAT.TestReporter.RESULT_FAILURE

464 		
465 		var arrayOfArrays = [manualSignatures, automatedSignatures];

467 		for (var i = 0; i < arrayOfArrays.length; i++) {
468 			var currentArray = arrayOfArrays[i];
469 			for (var j = 0; j < currentArray.length; j++) {

475 		reporter.addTest("testSomething.testIncomplete");
476 		
477 		reporter.collectResultsOfTests();
478 		assertEquals(reporter.numFailed, 2);
479 		assertEquals(reporter.numPassed, 2);
