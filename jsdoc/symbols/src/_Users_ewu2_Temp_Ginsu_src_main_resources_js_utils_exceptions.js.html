 12 /**
 13  * This is a helper function to handle the fact that JavaScript will sometimes 
 14  * throw strings as exceptions while at other times throw full fledged exception 
 15  * objects. It determines if a message is part of a given exception.
 16  *
 17  * @param {mixed} e The exception we are handling.
 18  * @param {string} message The message to check for
 19  * 
 20  * @return {boolean} true if message is part of the thrown exception.
 21  */
 22 IMAT.exceptionStartsWithMessage = function(e, message)
 23 {
 24 	var isPartofException = false;
 25 	if(typeof e == "string")
 26 	{
 27 		isPartofException = e.startsWith(message);
 28 	}
 29 	else if (e.message)
 30 	{
 31 		isPartofException = e.message.startsWith(message);
 32 	}
 33 	return isPartofException;
 34 };
