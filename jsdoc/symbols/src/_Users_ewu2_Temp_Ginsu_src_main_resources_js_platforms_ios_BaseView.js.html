 12 /**
 13  * @class The Base view is responsible for a few things...
 14  * <ol> 
 15  *		<li>It validates the state of the current view context while providing
 16  * 		a common logging patterns and screen capture.</li>
 17  *		<li>It validates the state of the current view context.</li> 
 18  *		<li>It holds references to the global application and target objects 
 19  * 		defined by Apple's UIAutomation framework.</li>
 20  * 		<li>Provides base functionality for all views to get access to elements
 21  * 		on the screen.</li>
 22  *		<li>Does common things that all other views miight need to do like 
 23  * 		Dismiss the keyboard or wait for some sort of network activity to 
 24  * 		complete.</li>
 25  * </ol>	
 26  *
 27  * All views that automate parts of the app should derive from this class.
 28  */

 61 	/**
 62 	 * Handles State validation logging and screen capturing leveraging a given 
 63 	 * validator function.
 64 	 * 
 65 	 * @param {string} stateName 
 66 	 * 					The state you want to validate
 67 	 * @param {boolean} isTransientState
 68 	 * 					true if the state you are validating is transient due to 
 69 	 * 					asynchronous changes in the app (eg. due to network 
 70 	 * 					callbacks). If the validation throws an exception when 
 71 	 * 					this true, the exception will be ignored and the 
 72 	 * 					automation will be allowed to move on. This can happen 
 73 	 * 					if the asynchronous change causes the app to transition 
 74 	 * 					out of that state before the validation can finish.
 75 	 * @param {IMAT.BaseView} view
 76 	 *  				The target view to validate. This will become the first
 77 	 *					parameter passed to the inner validation function passed
 78 	 *  				to this method (validatorFunc).
 79 	 * @param {function} validatorFunc 
 80 	 * 					The callback function defining the valid state. This 
 81 	 * 					function should expect a view as the first parameter of 
 82 	 * 					the function and, optionally, an args object which, if 
 83 	 * 					defined will be passed to validatorFunc as the second 
 84 	 * 					parameter.
 85 	 * @param {object} args 
 86 	 * 					Optional. If defined, they will be passed to the 
 87 	 * 					validator function for dynamic state validation.
 88 	 *
 89 	 * @throws An exception of type {@type String} if there is a script error.
 90 	 * @throws An exception of Type {@type IMAT.AssertionException} if an 
 91 	 * 		assertion does not pass. 
 92 	 */
 93 	validateState: function(stateName, isTransientState, view, validatorFunc, args)
 94 	{
 95 		var validationLogMessage = "VALIDATE " + this.viewName + ": " + stateName + " state.";
 96 		IMAT.log_debug(validationLogMessage);
 97 		this.target.captureScreenWithName("SCREEN: " + validationLogMessage + " " + new Date());
 98 		try {
 99 			if(typeof args == "undefined") {
100 				validatorFunc(view);
101 			} else {
102 				validatorFunc(view, args);
103 			}
104 		} catch (e) {
105 			if(isTransientState) {
106 				IMAT.log_debug("Transient State Validation Failed, most likely "+
107 					"due to an asynchronous state transition in the app. "+
108 					"Exception is being ignored. " + e);
109 			} else {
110 				var validationPrefix = "Validation Error: ";
111 				if (e instanceof IMAT.AssertionException) {
112 					throw new IMAT.AssertionException(validationPrefix + 
113 						e.toString());
114 				} else {
115 					throw "Test Script Error (Please Fix)- " + e;
116 				}
117 			}
118 		}
119 		IMAT.log_trace("Validation on " + this.viewName + ": " + stateName + 
120 			" state passed." );
121 	},

123 	/**
124 	 * Waits for activity indicators to finish before letting the automation 
125 	 * move on. This function assumes that the intended activity indicator to 
126 	 * wait on is the first one found below the parameter <code>elem</code>.
127 	 *
128 	 * @param {UIAElement} elem
129 	 * 						The element containing the UIAActivityIndicator 
130 	 * 						element to be waited on. If no visible UIAElements 
131 	 * 						are found, the automation will move on.
132 	 * @param {number} timeout
133 	 * 						The length of time to wait for activity before 
134 	 * 						throwing an exception.
135 	 *
136 	 * @throws an exception if elem is not a valid element or if the timeout 
137 	 * defined is not a number.
138 	 */
139 	waitForActivity: function(elem, timeout)
140 	{
141 		var activityIndicator;
142 		if(elem.checkIsValid() && typeof timeout == "number") {
143 			//find the first visible activity indicator
144 			IMAT.log_trace("Number of Activity indicators: " + elem.activityIndicators().length);
145 			for (var i = 0; i < elem.activityIndicators().length; i++) {
146 				IMAT.log_trace("checking activity indicator with name: " + elem.activityIndicators()[i].name());
147 				if(elem.activityIndicators()[i].checkIsValid() && elem.activityIndicators()[i].isVisible()) {
148 					IMAT.log_trace("found activity indicator");
149 					activityIndicator = elem.activityIndicators()[i];
150 				} else {
151 					IMAT.log_trace("Activity indicator is valid: " + elem.activityIndicators()[i].checkIsValid());
152 					IMAT.log_trace("Activity indicator is visible: " + elem.activityIndicators()[i].isVisible());
153 				}
154 			}
155 			if (activityIndicator) {
156 				this.target.pushTimeout(timeout);
157 				activityIndicator.waitForInvisible();
158 				this.target.popTimeout();
159 			} else {
160 				IMAT.log_debug("No Visible activity indicators were found. Automation will continue");
161 				IMAT.log_state();	
162 			}
163 		} else {
164 			throw "Exception while waiting for activity | elem is valid: " + 
165 				(elem.checkIsValid() ? "true":"false") + ", type of timeout: " + typeof timeout +
166 				". Elem should be valid and the timeout should be a number";
167 		}

207 	//This function is not documented on purpose as it is not to be used 
208 	//directly by a consuming project. Instead, the user should call one of the
209 	//functions that indirectly uses _retrieveElement. See this.getElementFromView,
210 	//this.getElement, this.isElementFromViewPresent and this.isElementPresent
211 	_retrieveElement: function(elementID, viewID) {
212 		var element = UIAElementNil;
213         if (IMAT.viewMap && IMAT.viewMapPrefix)
214         {
215             IMAT.log_trace("View name is: " + this.viewName + " viewID is: " + viewID);
216             IMAT.log_trace("Finding element: " + elementID);
217             if (!viewID) {
218                 viewID = this.viewName;
219             }
220             this.target.pushTimeout(5);
221             var locator = IMAT.viewMap[viewID][elementID];
222             IMAT.log_trace("the view map returned the following locator: " + locator);
223             if (typeof locator == "function") {
224             	IMAT.log_trace("Locator for " + elementID + " was a function.");
225             	locatorArgs = [];
226             	for (var argInd = 2; argInd < arguments.length; argInd++) {
227             		locatorArgs.push(arguments[argInd]);
228             	}
229             	element = locator.apply(this, locatorArgs);
230             } else if (locator instanceof Array) {
231             	IMAT.log_trace("Locator for " + elementID + " was an array.");
232             	var tempElement = undefined;
233             	var i = 0;
234             	for (i = 0; i < locator.length; i++) {
235             		IMAT.log_trace("evaluating: " + IMAT.viewMapPrefix + locator[i]);
236                     tempElement = eval(IMAT.viewMapPrefix + locator[i]);
237             		if (tempElement && !(tempElement instanceof UIAElementNil)){
238             			element = tempElement;
239             			break;
240             		}
241                     IMAT.log_trace("array locator not found in loop: " + IMAT.viewMapPrefix + locator[i]);
242             	}
243             } else if (typeof locator == "string") {
244             	IMAT.log_trace("Locator for " + elementID + " was a string.");
245             	element = eval(IMAT.viewMapPrefix + locator);
246             }
247             this.target.popTimeout();
248             IMAT.log_debug("Found " + element + " when searching for " + elementID);
249         } else {
250             IMAT.log_warning("Both IMAT.viewMap and IMAT.viewMapPrefix must be defined to " +
251             	"use IMAT.BaseView.getElement(). Cannot get element.");
252         }
253         return element;
254 	},

256     /**
257      * Get an element defined within the {@link IMAT.viewMap}. <b>Note</b>, the
258      * view map is defined within the project when it is generated via the IMAT 
259      * command line utility and contains a set of key-value pairs mapping the 
260      * keys to locator values. These values can be one of three types:
261      * <ul>
262      * <li>A {@link String} representing the location of the element, like<br/> 
263      * <code>refreshButton : ".navigationBar().buttons()[\"Refresh\"]",</code><br/>
264      * In this case, the string is evaluated and the function simply returns the result.</li>
265      * <li>An {@link Array} of {@link String}s representing possible locations 
266      * of the element, like <br/><code>refreshButton : 
267      * [".navigationBar().buttons()[\"Refresh\"]", ".navigationBar().buttons()[0]"],</code><br/>
268      * In this case, each element in the array is evaluated until the first 
269      * element is found and does not contain an undefined result or UIAElementNil.</li>
270      * <li>A {@link function} which returns the desired element.</li>
271      * </ul>
272      * More concretely, see the following example of a viewMap for an imaginary
273      * project. Note the different types of locators. For the locators that are
274      * of type {@type string}, the IMAT.viewMapPrefix is appended before evaluating
275      * them.
276      * 
277      * @example
278      * //viewMap.js for an imaginary project..
279      * IMAT.viewMapPrefix = "UIATarget.localTarget().frontMostApp().mainWindow()";
280      * IMAT.viewMap = {	
281      * 	ItemListView : {
282      * 		searchBar :  ".toolbars()[0].searchBars()[0]",
283      * 		clearSearchButton :  [
284      * 			".toolbars()[0].searchBars()[0].buttons().firstWithName(\"Clear text\")",
285      * 			".toolbars()[0].searchBars()[0].buttons()[0]"],
286      * 		listButton :  ".toolbars()[0].buttons()[1]",
287      * 		itemWithName: function(name) {
288      * 			return UIATarget.localTarget().frontMostApp().mainWindow().tableViews[0].cells()[name];
289      * 		},
290      * 	},
291      * 	...
292      * };
293      * 
294      * @param {string} elementID
295      * 						The name of the property from the view map for the 
296      * 						current view. When the value assigned to this 
297      * 						property is evaluated, it should return the desired
298      *						element currently on screen.
299      * @param {string} viewID
300      *                      The view within the view map to pull the element from.
301      *                      This is helpful during instances where you might want
302      *                      to inherit elements from a parent view, but don't want
303      *                      to repeat the elementID in every child view. Instead,
304      *                      you can call <code>getElement()</code> from the parent 
305      *                      view and specify the viewID of the more generic set of
306      *                      locators. This is an optional argument. When it is not
307      *                      set, viewID takes the same value as <code>this.viewName<code>
308      *
309      * @returns {@type UIAElement}
310      * 
311      * @requires IMAT.viewMap, A map of locators defining elements on 
312      * each view. IMAT.viewMapPrefix, the prefix to all string based locators. 
313      */
314     getElementFromView: function(elementID, viewID){
315 		var element = this._retrieveElement.apply(this, arguments);
316         if (element instanceof UIAElementNil) {
317         	IMAT.log_warning("Could not locate element: " + elementID + 
318         		" in view: " + this.viewName);
319         }
320         return element;
321 	},

334      * [".navigationBar().buttons()[\"Refresh\"]", ".navigationBar().buttons()[0]"],</code><br/>
335      * In this case, each element in the array is evaluated until the first

343      *

348      * 	ItemListView : {

364      * 						property is evaluated, it should return the desired

372     getElement: function(elementID) {
373         var locatorArgs = [elementID, this.viewName];
374         for (var argInd = 1; argInd < arguments.length; argInd++) {
375             IMAT.log_trace("Pushing Argument from getElement()");
376             locatorArgs.push(arguments[argInd]);
377         }
378         IMAT.log_trace("Passing the following Locator Args to "+
379         	"getElementFromView([" + locatorArgs + "])");
380         return this.getElementFromView.apply(this, locatorArgs);
381     },

383     /**
384 	 * 	Safely determine if an element is present within the view. This function
385 	 *  uses locators the same way that {@link IMAT.BaseView#getElement} does to
386 	 *  locate the element in question Also note that this function will not 
387 	 *  throw an exception if the element does not exist or the locator 
388 	 *  evaluates to some non-valid function.
389 	 *	
390 	 *  @param {string} elementID
391 	 *						The element ID of the viewMap locator which we wish
392 	 * 						To test for existence
393 	 *
394 	 *  @param {string} viewID
395      *                      The view within the view map to pull the element from.
396      *                      This is helpful during instances where you might want
397      *                      to inherit elements from a parent view, but don't want
398      *                      to repeat the elementID in every child view. Instead,
399      *                      you can call <code>getElement()</code> from the parent 
400      *                      view and specify the viewID of the more generic set of
401      *                      locators. This is an optional argument. When it is not
402      *                      set, viewID takes the same value as <code>this.viewName<code>
403 	 * 
404 	 *  @requires IMAT.viewMap 
405 	 * 						A map of locators defining elements on each view. 
406 	 * 						IMAT.viewMapPrefix, the prefix to all string based 
407      * 						locators.
408 	 *
409 	 *  @returns {@type boolean} true if the object mapped to the element within  
410 	 * 		the view map evaluates to true and is not an instance of 
411 	 * 		UIAElementNil.
412 	 *  
413 	 *  @see IMAT.BaseView#getElementFromView
414 	 */
415     isElementFromViewPresent: function(elementID, viewID) {
416     	var isElementPresent = false;
417     	var element = undefined;
418     	try {
419     		element = this._retrieveElement.apply(this, arguments);
420     		if (element && !(element instanceof UIAElementNil)) {
421     			isElementPresent = true;
422     		}
423     	} catch (e) {
424     		IMAT.log_info("An exception was thrown when trying to get element:"+
425     			" " + elementID + " from view: " + viewID);
426     	}
427     	return isElementPresent;
428     },

434 	 *  throw an exception if the element does not exist or the locator

437 	 *  @param {string} elementID

439 	 * 						To test for existence
440 	 *  @requires IMAT.viewMap 
441 	 * 						A map of locators defining elements on each view.

457         }
458         return this.isElementFromViewPresent.apply(this, locatorArgs);

461 	/**
462 	 * 	Do nothing for the desired number of seconds.
463 	 *	@param {Number} seconds
464 	 *						The number of seconds to wait. If a string is pased,
465 	 *						will attempt to convert to an integer using parseInt.
466 	 *						Currently, only decimal numbers are allowed. If the 
467 	 *  					parameter parsed gives NaN, a warning is logged and 
468 	 *						seconds is set to the value 0. 
469 	 *  @returns this view.
470 	 */
471 	waitAction: function(seconds) {
472 		if (typeof seconds == "string")
473 		{
474 			var givenSeconds = seconds;
475 			seconds = parseInt(seconds, 10);
476 			if (isNaN(seconds))
477 			{
478 				seconds = 0;
479 				IMAT.log_warning("While calling waitAction, the given number of "+
480 				"seconds could not be parsed from a string into a number. Given " +
481 				"number of seconds was: " + givenSeconds);
482 			}
483 		}
484 		this.target.delay(seconds);
485 		return this;
486 	},
