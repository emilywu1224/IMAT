 12 /**
 13  * Returns the index within the array of the first element with its name
 14  * matching the supplied elementName.
 15  *
 16  * @param {UIAElementArray} elementArray 
 17  * 						The array containing the elements to search.
 18  * @param {string} name 
 19  * 						The name of the element to find. NOTE: This is case 
 20  * 						sensitive.
 21  *
 22  * @return {number} The whole number representing the index of the element 
 23  * within the array. If not found, this element will be undefined.
 24  */
 25 IMAT.indexOfFirstWithName = function(elementArray, name)
 26 {
 27 	var i = 0;
 28 	var index = undefined;
 29 	for (i = 0; i < elementArray.length; i++)
 30 	{
 31 		var currentElement = elementArray[i];
 32 		var elementIsNull = (currentElement === null || typeof currentElement === "undefined" || currentElement instanceof UIAElementNil);
 33 		if (!elementIsNull && currentElement.name() == name)
 34 		{
 35 			index = i;
 36 			break;
 37 		}
 38 	}
 39 	return index;
 40 };
