  6 var Class = (function() {
  7   var __extending = {};
  8 
  9   return {
 10     extend: function(parent, def) {
 11       if (arguments.length == 1) { def = parent; parent = null; }
 12       var func = function() {
 13         if (arguments[0] ==  __extending) { return; }
 14         this.initialize.apply(this, arguments);
 15       };
 16       if (typeof(parent) == 'function') {
 17         func.prototype = new parent( __extending);
 18       }
 19       var mixins = [];
 20       var clobberMixins = [];
 21       if (def && def.include) {
 22         if (def.include.reverse) {
 23           // methods defined in later mixins should override prior
 24           mixins = mixins.concat(def.include.reverse());
 25         } else {
 26           mixins.push(def.include);
 27         }
 28         delete def.include; // clean syntax sugar
 29       }
 30       if(def && def.overrideInclude) {
 31     	  if (def.overrideInclude.reverse) {
 32               // methods defined in later mixins should override prior
 33     		  clobberMixins = clobberMixins.concat(def.overrideInclude.reverse());
 34           } else {
 35         	  clobberMixins.push(def.overrideInclude);
 36           }
 37     	  delete def.overrideInclude; // clean syntax sugar
 38       }
 39       if (def) { 
 40       	Class.inherit(func.prototype, def); 
 41       }
 42       for (var i = 0; (mixin = mixins[i]); i++) {
 43     	  Class.mixin(func.prototype, mixin);
 44       }
 45       // similar to the above loop that carefully imports mixins, this version aggresively clobbers functions
 46       for (var j = 0; (clobberMixin = clobberMixins[j]); j++) {
 47           Class.mixin(func.prototype, clobberMixin, true);
 48       }
 49       return func;
 50     },
 51     mixin: function (dest, src, clobber) {
 52       clobber = clobber || false;
 53       if (typeof(src) != 'undefined' && src !== null) {
 54         for (var prop in src) {
 55           // if clobbering don't clobber initialize
 56           if ((clobber && prop != 'initialize') || (!dest[prop] && typeof(src[prop]) == 'function')) {
 57         	  dest[prop] = src[prop];
 58           }
 59         }
 60       }
 61       return dest;
 62     },
 63     inherit: function(dest, src, fname) {
 64       if (arguments.length == 3) {
 65         var ancestor = dest[fname], descendent = src[fname], method = descendent;
 66         descendent = function() {
 67           var ref = this.parent; this.parent = ancestor;
 68           var result = method.apply(this, arguments);
 69           if (ref) { 
 70           	this.parent = ref;
 71           } else {
 72           	delete this.parent;
 73           }
 74           return result;
 75         };
 76         // mask the underlying method
 77         descendent.valueOf = function() { return method; };
 78         descendent.toString = function() { return method.toString(); };
 79         dest[fname] = descendent;
 80       } else {
 81         for (var prop in src) {
 82           if (dest[prop] && typeof(src[prop]) == 'function') {
 83             Class.inherit(dest, src, prop);
 84           } else {
 85             dest[prop] = src[prop];
 86           }
 87         }
 88       }
 89       return dest;
 90     },
 91     singleton: function() {
 92       var args = arguments;
 93       if (args.length == 2 && args[0].getInstance) {
 94         var klass = args[0].getInstance(__extending);
 95         // we're extending a singleton swap it out for it's class
 96         if (klass) { args[0] = klass; }
 97       }
 98 
 99       return (function(args){
100         // store instance and class in private variables
101         var instance = false;
102         var klass = Class.extend.apply(args.callee, args);
103         return {
104           getInstance: function () {
105             if (arguments[0] == __extending) {
106             	return klass;
107             }
108             if (instance) { 
109             	return instance;
110             }
111             return (instance = new klass());
112           }
113         };
114       })(args);
115     }
116   };
117 })();
118 
119 // finally remap Class.create for backward compatability with prototype
120 Class.create = function() {
121   return Class.extend.apply(this, arguments);
122 };
