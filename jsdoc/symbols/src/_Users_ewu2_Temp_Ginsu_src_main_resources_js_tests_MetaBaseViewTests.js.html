 12 //Core test framework
 13 /**
 14  * This class Tests the BaseView class to ensure that its functions work as 
 15  * expected
 16  * @ignore 
 17  */
 18 IMAT.MetaBaseViewTests = Class.extend(IMAT.BaseTestSet, {
 19 	
 20 	title: "MetaBaseViewTests",
 21 	/**
 22 	 * Initializes the MetaTest Class object.
 23 	 */
 24 	initialize: function() {
 25 		this.baseView = new IMAT.MockView();
 26 	},
 27 	
 28 	/**
 29 	 * setup the App to normalize the way tests are run. Initialize the view context and do whatever
 30 	 * we can to clear out the old set of test data and set the right target server environment
 31 	 */
 32 	setUp: function() {
 33 		//Do nothing
 34 	},
 35 	
 36 	/**
 37 	 * tearDown the App to normalize the way tests finish. Clear the cache if possible.
 38 	 */
 39 	tearDown: function() {
 40 		//Do nothing
 41 	},
 42 	
 43 	setUpTestSet: function() {
 44 		//Do nothing.
 45 	},
 46 	
 47 	tearDownTestSet: function()	{
 48 		//Do nothing
 49 	},
 50 	
 51 	doCleanup: function() {
 52 		//Do nothing
 53 	},
 54 	
 55 	testThatValidateStateReturnsNormallyForPassingAssertions: function() {
 56 		try {
 57 			this.baseView.validateHappyCase();
 58 		} catch (e) {
 59 			assertTrue(false, "Validating the happy case failed. with exception" +
 60 				e.toString());
 61 		}
 62 	},
 63 	
 64 	testThatValidateStateThrowsAnAssertionExceptionForFailedAssertions: function() {
 65 		try {
 66 			this.baseView.validateWithForcedAssertionException();
 67 			assertTrue(false, "we expected a forced IMAT.AsserrtionException");
 68 		} catch (e) {
 69 			assertTrue(e instanceof IMAT.AssertionException, 
 70 				"Validate State did not throw an assertion exception for "+
 71 				"failed validation. Instead we got: " + e.toString());
 72 		}
 73 	},
 74 	
 75 	testThatValidateStateThrowsAnExceptionForGeneralExceptions: function() {
 76 		try {
 77 			this.baseView.validateWithForcedException();
 78 			assertTrue(false, "we expected a forced exception");
 79 		} catch (e) {
 80 			assertFalse(e instanceof IMAT.AssertionException, 
 81 				"Validate State threw an assertion exception for a general "+
 82 				"exception");
 83 		}
 84 	},
 85 	
 86 	testThatValidateStateDoesNotThowExceptionsForTransientStates: function() {
 87 		try {
 88 			this.baseView.validateTransientStateWithForcedAssertionException();
 89 		} catch (e) {
 90 			assertTrue(false, "transient states should only show warnings "+
 91 				"and never pass on assertion exceptions");
 92 		}
 93 		
 94 		try {
 95 			this.baseView.validateTransientStateWithForcedException();
 96 		} catch (e2) {
 97 			assertTrue(false, "transient states should only show warnings "+
 98 				"and never pass on exceptions");
 99 		}
100 	},
101 	
102 	testThatValidateStateDoesNotRequireDynamicArgsInValidatorFunction: function() {
103 		this.baseView.validateAnonymousFuncRequiresNoArgs();
104 	},
105 	
106 	testThatValidateStateAcceptsOptionalArgsObjInValidatorFunction: function() {
107 		this.baseView.validateAnonymousFuncTakesOptionalArgsObj();
108 	}
109 });
110 
111 IMAT.suiteRunner.addTestSet( new IMAT.MetaBaseViewTests());
