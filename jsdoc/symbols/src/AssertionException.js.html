 12 /**
 13  * 	Constructs an IMAT.AssertionException object which inherits from the 
 14  *  JavaScript ERROR object. 
 15  * @class The {@link IMAT.AssertionException} object is thrown any time the a
 16  *  test failure is thrown due to an assertion API. Throwing this object will 
 17  *  cause the current phase in the test execution lifecycle to be marked as a 
 18  *  failure instead of an issue.
 19  * @extends <a href=https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Error>Error</a>
 20  * @memberOf IMAT 
 21  */
 22 IMAT.AssertionException = function(message)
 23 {
 24 	this.prototype = Error.prototype;
 25 	this.name = "AssertionException";
 26 	this.message = (message) ? message : "The assertion was not valid.";
 27 	this.toString = function() {
 28 		return this.message;
 29 	};
 30 };
