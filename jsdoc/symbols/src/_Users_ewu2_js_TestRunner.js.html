 12 /**
 13  * 	Constructs a TestRunner object
 14  *	@class The {@link IMAT.TestRunner} object is the workhorse of the test
 15  *	execution lifecycle of a given Test Set. It's major responsibilities 
 16  *	include:
 17  *  <ul>
 18  *		<li>Inspecting a Test Set to determine which properties are
 19  * 		runnable testcases.</li>
 20  *		<li>Filtering out tests which do not match the set of internal 
 21  * 		filters it currently holds</li>
 22  *  	<li>Running tests which match the filters it currently holds</li>
 23  *  </ul>
 24  * @memberOf IMAT 
 25  */
 26  /*
 27   * @TODO implement an optional retry mechanism to gain more validation when
 28   *	tests fail.
 29   */
 30 IMAT.TestRunner = Class.extend(/** @lends IMAT.TestRunner# */{
 31 	
 32 	/**
 33 	 * The constructor called when invoking "new" on this class object.
 34 	 * @ignore 
 35 	 */
 36 	initialize: function()
 37 	{
 38 		this.filtersArray = [];
 39 	},
 40 	
 41 	/**
 42 	 * Returns the signature for the test case
 43 	 * 
 44 	 * @param {IMAT.BaseTestSet} testSet
 45 	 * 					A testSet in our test harness
 46 	 * @param {string} testCaseName 
 47 	 * 					The name of the test case to be run.
 48 	 * 
 49 	 * @return {string} The signature of the test case. This will be of the form
 50 	 * <code>SomeTestSet.testCaseName()</code> where SomeTestSet is the test set
 51 	 *  class and testCaseName is the method representing the test case.
 52 	 */
 53 	getTestCaseSignature: function(testSet, testCaseName)
 54 	{
 55 		return testSet.title + "." + testCaseName + "()";
 56 	},
 57 	
 58 	/**
 59 	 * Determines if a given property is a function we should run as part of the 
 60 	 * test
 61 	 * 
 62 	 * @param {mixed} prop
 63 	 * 					A property of the test object which may or may not be a 
 64 	 * 					test case
 65 	 * @param {IMAT.BaseTestSet} testSet
 66 	 * 					A testSet in our test harness
 67 	 * 
 68 	 * @return {boolean} true if prop is a function and starts with word "test" 
 69 	 * like "testSomething: function(){...},"
 70 	 */
 71 	isPropValidTestCase: function(prop, testSet)
 72 	{
 73 		var funcPattern = /^test[0-9,a-z,A-Z,_]+/;
 74 		return typeof testSet[prop] == "function" && funcPattern.test(prop.toString());
 75 	},
 76 	
 77 	/**
 78 	 * Determines if a test case matches at least one of the filters given with 
 79 	 * the test set.
 80 	 * 
 81 	 * @param {string} testCase
 82 	 * 					The name of the test case to be run.
 83 	 * @param {IMAT.BaseTestSet} testSet
 84 	 * 					A testSet in our test harness
 85 	 * 
 86 	 * @return {boolean} true if the testCase matches at least one of the filters.
 87 	 */
 88 	testCaseMatchesFilter: function(testCase, testSet)
 89 	{
 90 		var filterPattern;
 91 		var isMatch = false;
 92 		var testSignature = this.getTestCaseSignature(testSet, testCase);
 93 		if(this.filtersArray.length > 0)
 94 		{
 95 			for(var i = 0; i < this.filtersArray.length; i++)
 96 			{
 97 				filterPattern = new RegExp(this.filtersArray[i]);
 98 				if (filterPattern.test(testSignature))
 99 				{
100 					isMatch = true;
101 					break;
102 				}
103 			}
104 		}
105 		else
106 		{
107 			isMatch = true;
108 		}
109 		return isMatch;
110 	},
111 	
112 	/**
113 	 * Convenience function for determining if the prop is a test in the testSet 
114 	 * that should be run. This function validates that prop is both a valid 
115 	 * test function and that it matches at least one filter.
116 	 * 
117 	 * @param {mixed} prop
118 	 * 					A property of the test object which may or may not be a 
119 	 * 					test case
120 	 * @param {IMAT.BaseTestSet} testSet
121 	 * 					A testSet in our test harness
122 	 *
123 	 * @returns {boolean} true if the test should be run.
124 	 */
125 	isPropRunnableTest: function(prop, testSet)
126 	{
127 		return (this.isPropValidTestCase(prop, testSet) && 
128 			this.testCaseMatchesFilter(prop, testSet));
129 	},
130 	
131 	/**
132 	 * Determine if a Test Set contains runnable test cases.
133 	 *
134 	 * @param {IMAT.BaseTestSet} testSet.
135 	 * 					A testSet in our test harness
136 	 *
137 	 * @returns {boolean} true if the testSet has runnable test cases.
138 	 */
139 	containsRunnableTestCases: function(testSet)
140 	{		
141 		var isRunnable = false;
142 		for(prop in testSet)
143 		{
144 			if (this.isPropRunnableTest(prop, testSet)) 
145 			{
146 				isRunnable = true;
147 				break;
148 			}
149 		}
150 		return isRunnable;
151 	},
152 	
153 	/**
154 	 * Returns an array of the runnable tests in the testSet. 
155 	 *
156 	 * @param {IMAT.BaseTestSet} testSet.
157 	 * 					A Test Set which we may execute against.
158 	 */
159 	collectRunnableTests: function(testSet, runnableTests)
160 	{
161 		if (!runnableTests) {
162 			runnableTests = [];
163 		}
164 		for(prop in testSet) {
165 			if (this.isPropRunnableTest(prop, testSet))  {
166 				runnableTests.push(this.getTestCaseSignature(testSet, prop));
167 			}
168 		}
169 	},
170 	
171 	/**
172 	 * Logs out a list of the runnable tests in the testSet. This is a function 
173 	 * written for test development convenience to see if your filters are 
174 	 * including the appropriate tests.
175 	 *
176 	 * @param {IMAT.BaseTestSet} testSet.
177 	 * 					A Test Set which we may execute against.
178 	 */
179 	previewRunnableTests: function(testSet)
180 	{
181 		IMAT.log_debug("\tTestSet: " + testSet.title);
182 		for(prop in testSet)
183 		{
184 			if (this.isPropRunnableTest(prop, testSet)) 
185 			{
186 				IMAT.log_debug("\t\t" + testSet.title + "." + prop + "()");
187 			}
188 		}
189 	},
190 	
191 	/**
192 	 * Runs a given test case in the test set. Never call this function 
193 	 * directly. Its merely a helper for {@link IMAT.TestRunner.runTestCases}
194 	 * 
195 	 * @param {IMAT.BaseTestSet} testSet.
196 	 * 					A Test Set which we will execute.
197 	 * @param {string} testCase
198 	 * 					The name of the test case we are about to run.
199 	 */
200 	runTestCase: function(testSet, testCaseName) {
201 		var testSignature = this.getTestCaseSignature(testSet, testCaseName);
202 		
203 		//this try block allows us our "fail fast" approach
204 		try {
205 			IMAT.log_start(testSignature);
206 			if (!IMAT.settings.SKIP_SET_UP_TEST_CASE) {
207 				testSet.setUp();
208 			}
209 			testSet[testCaseName]();
210 			if (!IMAT.settings.SKIP_TEAR_DOWN_TEST_CASE) {
211 				testSet.tearDown();
212 			}
213 			//if we get here, the test has passed.
214 			IMAT.log_pass(testSignature);
215 			this.markTestAsRanWithResult(testSignature, IMAT.TestReporter.RESULT_PASS);
216 		} catch (e)  {
217 			this.logException(e, testSignature);
218 			//Invoke the cleanup process so that we can make an attempt at  
219 			//starting the next test fresh
220 			if (!IMAT.settings.SKIP_DO_CLEANUP) {
221 				testSet.doCleanup();
222 			}
223 		}
224 	},
225 	
226 	/**
227 	 * Runs a set of test cases. This is the main function of the TestRunner 
228 	 * class. It drives each test set through its testing lifecycle.
229 	 *
230 	 * @param {IMAT.BaseTestSet} testSet.
231 	 * 					A Test Set which we will execute.
232 	 * 
233 	 * @param {function} test
234 	 *  				The test we are about to run.
235 	 */
236 	runTestCases: function(testSet, test)
237 	{		
238 		if(this.containsRunnableTestCases(testSet))
239 		{
240 			// NOTE: The Code which translates the Plist files over to Junit
241 			// XML reports keys off these tokens to do the translation. 
242 			// In terms of guaranteeing that the report can be generated it is
243 			// important to ensure that these tokens are always printed out at
244 			// the beginning and end of each testSet.
245 			var setUpToken = testSet.title + ".setUpTestSet();";
246 			var tearDownToken = testSet.title + ".tearDownTestSet();";
247 			try {
248 				IMAT.log_start(setUpToken);
249 				if (!IMAT.settings.SKIP_SET_UP_TEST_SET ) {
250 					testSet.setUpTestSet();
251 				} 
252 				IMAT.log_pass(setUpToken);
253 			} catch (setUpException) {
254 				this.logException(setUpException, setUpToken);
255 			}
256 			for(prop in testSet)
257 			{
258 				// NOTE: We do not check for hasOwnProperty() because that  
259 				// function does not check the prototype of the object, just the
260 				// object itself
261 				if (this.isPropRunnableTest(prop, testSet)) 
262 				{
263 					this.runTestCase(testSet, prop);
264 				}
265 			}
266 			try {
267 				IMAT.log_start(tearDownToken);
268 				if (!IMAT.settings.SKIP_TEAR_DOWN_TEST_SET) {
269 					testSet.tearDownTestSet();
270 				}
271 				IMAT.log_pass(tearDownToken);
272 			} catch (tearDownException) {
273 				this.logException(tearDownException, tearDownToken);
274 			}
275 		}
276 	},
277 	
278 	/**
279 	 * Maintain an internal reference to the array of filters so that the set of
280 	 * runnable tests can be generated. Users should never call this method 
281 	 * directly.
282 	 *
283 	 * @param {array} the Array of strings to use as filters.
284 	 * 
285 	 * @see IMAT.SuiteRunner#previewAllRunnableTests for the full set of 
286 	 * documentation.
287 	 */
288 	addFilters : function(filtersArray)
289 	{
290 		this.filtersArray = this.filtersArray.concat(filtersArray);
291 	},
292 	
293 	/**
294 	 * Handle the exception from a failure in the execution of the test life-
295 	 * cycle. Note, this is used internally by The testing framework and should
296 	 * not be used directly in an automation project.
297 	 * 
298 	 * This function will extract the message from the exception object 
299 	 * if it derives from the JavaScript Error object and do everything it can
300 	 * to ensure that the exception is converted to a string before loging out 
301 	 * to the results log. Furthermore, this function will handle the logging of
302 	 * the current state of the application shown in view and mark the test as 
303 	 * either failed or needing attention depending on the type of exception 
304 	 * thrown. If the exception comes from one of the IMAT assertion APIs
305 	 * (e.g. assertTrue()) then the test will be marked as failing. However, if
306 	 * there is any other type of exception, the exception will be logged as an
307 	 * issue so that the testing engineers known that there is a problem in the 
308 	 * scripts.
309 	 *
310 	 * @param {mixed} exception 
311 	 * 						The exception to be handled.
312 	 * @param {srting} signature
313 	 * 						the signature of the function where the exception  
314 	 * 						came from.
315 	 * 
316 	 * @see IMAT.settings.LOG_STATE_ON_ERROR if you would like to turn off 
317 	 * logging the state of the application when an error happens. 
318 	 * @see IMAT.AssertionException for more information on the exception 
319 	 *  object which will report as test failures.
320 	 */
321 	logException: function(exception, signature)
322 	{
323 		//first get the error string and log it out.
324 		var errorString = "";
325 		if (typeof exception == "string") {
326 			errorString = exception;
327 		} else if (exception.message) {
328 			errorString = exception.message;
329 		} else {
330 			errorString = exception.toString();
331 		}
332 		IMAT.log_error(errorString);
333 		
334 		//then print out the current view tree
335 		if (IMAT.settings.LOG_STATE_ON_ERROR) {
336 			IMAT.log_state();
337 		}
338 		//classify the test as having failed or needing attention
339 		if (exception instanceof IMAT.AssertionException) {
340 			IMAT.log_fail(signature);
341 			this.markTestAsRanWithResult(signature, IMAT.TestReporter.RESULT_FAILURE);
342 		} else {
343 			IMAT.log_issue(signature);
344 			this.markTestAsRanWithResult(signature, IMAT.TestReporter.RESULT_ERROR);
345 		}
346 	},
347 	
348 	setTestReporter: function(testReporter) 
349 	{
350 		this.reporter = testReporter;
351 	},
352 	
353 	markTestAsRanWithResult: function(testSignature, testResult) 
354 	{
355 		if (this.reporter) {
356 			this.reporter.markTestAsRanWithResult(testSignature, testResult);
357 		}
358 	}
359 });
