 12 //Core test framework
 13 /**
 14  * This class Tests the BaseView class to ensure that its functions work as 
 15  * expected
 16  * @ignore 
 17  */
 18 IMAT.MockView = Class.extend(IMAT.BaseView, {
 19 	
 20 	title: "MockView",
 21 	/**
 22 	 * Initializes the MetaTest Class object.
 23 	 */
 24 	initialize: function() {
 25 		this.parent();
 26 		this.viewName = "MockView";
 27 	},
 28 	
 29 	validateHappyCase: function() {
 30 		this.validateState("happy case", false, null, function(){
 31 			assertTrue(true);
 32 		});
 33 	},
 34 	
 35 	validateWithForcedAssertionException: function() {
 36 		this.validateState("throwing an AssertionException", false, null, function(){
 37 			throw new IMAT.AssertionException("Forced Assertion Exception");
 38 		});
 39 	},
 40 	
 41 	validateWithForcedException: function() {
 42 		var foo = {};
 43 		this.validateState("throwing an Exception", false, null, function(){
 44 			foo.jones(); //this function does not exist
 45 		});
 46 	},
 47 	
 48 	validateTransientStateWithForcedAssertionException: function() {
 49 		this.validateState("throwing an AssertionException", true, null, function(){
 50 			throw new IMAT.AssertionException("Forced Assertion Exception");
 51 		});
 52 	},
 53 	
 54 	validateTransientStateWithForcedException: function() {
 55 		var foo = {};
 56 		this.validateState("throwing an Exception w/ transient state", true, null, function(){
 57 			foo.jones(); //this function does not exist
 58 		});
 59 	},
 60 	
 61 	validateAnonymousFuncRequiresNoArgs: function() {
 62 		this.validateState("anonymous func reqires no args", true, null, function(){
 63 			//do nothing
 64 			assertTrue(arguments.length === 1);
 65 		});
 66 	},
 67 	
 68 	validateAnonymousFuncTakesOptionalArgsObj: function() {
 69 		this.validateState("anonymous func takes optional args", true, null, function(view, args){
 70 			assertEquals(args.ima, "baller");
 71 		}, {ima:"baller"});
 72 	}
 73 });
