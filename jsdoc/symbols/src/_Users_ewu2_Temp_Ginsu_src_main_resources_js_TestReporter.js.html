 12 /**
 13  * Creates and Maintains Reporting Statistics 
 14  * @class The {@link IMAT.TestReporter} object synthesises statistics and 
 15  * maintains data to keep track of test counts, pass/failure rates, and will
 16  * report on the number of manual vs. automated tests there are in a code base.
 17  */
 18 IMAT.TestReporter = Class.extend(/** @lends IMAT.TestReporter# */{
 19 	
 20 	/**
 21 	 * The constructor called when invoking "new" on this class object.
 22 	 * @ignore 
 23 	 */
 24 	initialize: function()
 25 	{
 26 		this.automatedTests = [];
 27 		this.manualTests = [];
 28 		this.incompleteTests = [];
 29 		this.numFailed = 0;
 30 		this.numErrored = 0;
 31 		this.numPassed = 0;
 32 	},
 33 	
 34 	/**
 35 	 * 
 36 	 */
 37 	reportProjection: function() 
 38 	{
 39 		var numAutomatedTests = this.automatedTests.length;
 40 		var numManualTests = this.manualTests.length; 
 41 		var percentAutomated = undefined;
 42 		if (numAutomatedTests + numManualTests > 0) {
 43 			percentAutomated = numAutomatedTests / (numAutomatedTests + numManualTests) * 100;
 44 			IMAT.log_info("Number of Automated Tests:\t" + numAutomatedTests);
 45 			IMAT.log_info("Number of Manual Tests:\t\t" + numManualTests);
 46 			IMAT.log_info("Percentage automated:\t\t" + percentAutomated + "%");
 47 			IMAT.log_info("****Expected Tests****");
 48 			this.listTests(this.automatedTests);
 49 			this.listTests(this.manualTests);
 50 			IMAT.log_info("**********************");
 51 		} else {
 52 			IMAT.log_info("There were no tests to report against.");
 53 		}
 54 	},
 55 	
 56 	reportRetrospective: function()
 57 	{
 58 		this.collectResultsOfTests();
 59 		var numAutomatedTests = this.automatedTests.length;
 60 		var numManualTests = this.manualTests.length; 
 61 		var totalPossibleTests = numAutomatedTests + numManualTests;
 62 		if ( totalPossibleTests > 0) {
 63 			var passPercentage = this.numPassed / totalPossibleTests * 100;
 64 			IMAT.log_info("Number of Tests passed:\t\t" + this.numPassed);
 65 			IMAT.log_info("Number of Tests failed:\t\t" + this.numFailed);
 66 			IMAT.log_info("Number of Tests with errors:\t" + this.numErrored);
 67 			IMAT.log_info("Pass Percentage:\t\t\t" + passPercentage + "%");
 68 			if (this.incompleteTests.length > 0) {
 69 				var completionPercentage = this.incompleteTests.length / totalPossibleTests * 100;
 70 				IMAT.log_info("Number of Tests that did not run: " + this.incompleteTests.length);
 71 				IMAT.log_info("Completion Percentage: " + completionPercentage);
 72 				IMAT.log_info("****Tests that did not get run.****");
 73 				this.listTests(this.incompleteTests);
 74 				IMAT.log_info("**********************");
 75 			} else {
 76 				IMAT.log_info("Completion Percentage:\t\t100%");
 77 			}
 78 		} else {
 79 			IMAT.log_info("There were no tests to report against.");
 80 		}
 81 	},
 82 	
 83 	
 84 	addTest: function(testSignature)
 85 	{
 86 		var test = {};
 87 		test.hasRun = false;
 88 		test.result = undefined;
 89 		if (this.isManualTest(testSignature)) {
 90 			this.manualTests[testSignature] = test;
 91 			this.manualTests.length++;
 92 		} else {
 93 			IMAT.log_trace("Adding Automated test: " + testSignature);
 94 			this.automatedTests[testSignature] = test;
 95 			this.automatedTests.length++;
 96 		}
 97 	},
 98 	
 99 	markTestAsRanWithResult: function(testSignature, testResult) 
100 	{
101 		if (this.isManualTest(testSignature) && this.manualTests[testSignature]) {
102 			this.manualTests[testSignature].hasRun = true;
103 			this.manualTests[testSignature].result = testResult;
104 		} else if (this.automatedTests[testSignature]) {
105 			this.automatedTests[testSignature].hasRun = true;
106 			this.automatedTests[testSignature].result = testResult;
107 		}
108 	},
109 	
110 	/**
111 	 * Determine if a test is a manual test based on it's signature. A test is a
112 	 * manual test if the function name starts with "testManually"
113 	 *  
114 	 * 	@param {String}
115 	 *					The signature of the test. 
116 	 *
117 	 *  @returns {boolean} true if the test is manual, false otherwise
118 	 *  @see  IMAT.TestRunner#getTestCaseSignature
119 	 */
120 	isManualTest: function(testSignature)
121 	{
122 		var manualTestPattern = /^[0-9,a-z,A-Z,_]+\.testManually[0-9,a-z,A-Z,_]+/;
123 		var result = false;
124 		if (testSignature && typeof testSignature === "string") {
125 			result = manualTestPattern.test(testSignature);
126 		}
127 		IMAT.log_trace("Is " + testSignature + " a manual test signature?: " + 
128 			(result ? "YES":"NO"));
129 		return result;
130 	},
131 	
132 	/*
133 	 * Log the test signatures in one of the test arrays of this object.
134 	 * @ignore 
135 	 *  
136 	 * 	@param {Array}
137 	 *					The array of test signatures 
138 	 * 
139 	 */
140 	listTests: function(testArray) 
141 	{
142 		for (testSignature in testArray) 
143 		{
144 			if (testArray.hasOwnProperty(testSignature)) {
145 				IMAT.log_info(testSignature);
146 			}
147 		}
148 	},
149 	
150 	collectResultsOfTests: function()
151 	{
152 		var numberFailed = 0;
153 		var numberPassed = 0;
154 		var numberErrored = 0;
155 		var incompleteTests = [];
156 		this.iterateOverTestsWithCallback(function(test, signature){
157 			if (test.hasRun) {
158 				if (test.result === IMAT.TestReporter.RESULT_PASS) {
159 					numberPassed++;
160 				} else if (test.result === IMAT.TestReporter.RESULT_FAILURE) {
161 					numberFailed++;
162 				} else if (test.result === IMAT.TestReporter.RESULT_ERROR) {
163 					numberErrored++;
164 				}
165 			} else {
166 				incompleteTests[signature] = test;
167 				incompleteTests.length++;
168 			}		
169 		});
170 		this.numFailed = numberFailed;
171 		this.numErrored = numberErrored;
172 		this.numPassed = numberPassed;
173 		this.incompleteTests = incompleteTests;
174 	},
175 	
176 	/**
177 	 * 
178 	 */
179 	iterateOverTestsWithCallback: function(callback)
180 	{
181 		for (autoTest in this.automatedTests) {
182 			if (this.automatedTests.hasOwnProperty(autoTest)) {
183 				callback(this.automatedTests[autoTest], autoTest);
184 			}
185 		}
186 		for (manualTest in this.manualTests) {
187 			if (this.manualTests.hasOwnProperty(manualTest)) {
188 				callback(this.manualTests[manualTest], manualTest);
189 			}
190 		}
191 	}
192 });
193 
194 IMAT.TestReporter.RESULT_ERROR = "ERROR";
195 IMAT.TestReporter.RESULT_FAILURE = "FAILURE";
196 IMAT.TestReporter.RESULT_PASS = "PASS";
