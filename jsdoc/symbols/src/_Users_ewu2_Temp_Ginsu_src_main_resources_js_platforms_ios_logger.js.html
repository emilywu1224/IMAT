 12 /**
 13  * Logs an error if the log level matches. Will take a screenshot if the tests are running on the
 14  * device.
 15  *
 16  * @param message string - The message to be logged
 17  */
 18 IMAT.log_error = function(message)
 19 {
 20 	if (IMAT.settings.logLevel >= IMAT.logLevels.LOG_ERROR)
 21 	{
 22 		//NOTE: this will take a screenshot as well.
 23 		UIALogger.logError(message);
 24 	}
 25 };
 26 
 27 /**
 28  * Logs a warning if the log level matches. Will take a screenshot if the tests are running on the

 42 /**
 43  * Logs an info message if the log level matches.
 44  *
 45  * @param message string - The message to be logged
 46  */
 47 IMAT.log_info = function(message)
 48 {
 49 	if (IMAT.settings.logLevel >= IMAT.logLevels.LOG_INFO)
 50 	{
 51 		UIALogger.logMessage(message);
 52 	}
 53 };

 55 /**
 56  * Logs a debug message if the log level matches.
 57  *
 58  * @param message string - The message to be logged
 59  */
 60 IMAT.log_debug = function(message)
 61 {
 62 	if (IMAT.settings.logLevel >= IMAT.logLevels.LOG_DEBUG)
 63 	{
 64 		UIALogger.logDebug("DEBUG: " + message);
 65 	}
 66 };

 68 /**

 75 	if(IMAT.settings.logLevel >=  IMAT.logLevels.LOG_TRACE)
