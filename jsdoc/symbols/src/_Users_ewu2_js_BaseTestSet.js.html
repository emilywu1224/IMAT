 12 /**
 13  * Constructs a {@link IMAT.BaseTestSet}. 
 14  * 
 15  * @class The {@link IMAT.BaseTestSet} is the most basic of all tests and 
 16  * does nothing more than provide stubbed out methods that the 
 17  * {@link IMAT.TestRunner} expects and initialize some basic instance 
 18  * variables related to its title etc. 
 19  */
 20 IMAT.BaseTestSet = Class.extend(/** @lends IMAT.BaseTestSet.prototype */{
 21 	
 22 	/**
 23 	 * The Title this test goes by. Used to identify this test in the logs. 
 24 	 * 
 25 	 * @type String 
 26 	 */
 27 	title: "Base Test Class",
 28 	
 29 	/**
 30 	 * A field that tells the TestRunner object to capture the state of the AUT
 31 	 * when a test fails. When truthy, the {@link IMAT.TestRunner} class will 
 32 	 * call <code>IMAT.log_state()</code> after a test fails. When falsy, the 
 33 	 * {@link IMAT.TestRunner} will not capture any state.
 34 	 * 
 35 	 * @type boolean
 36 	 */
 37 	logOnFailure: true,
 38 	
 39 	//The constructor called when invoking "new" on this class object.
 40 	//See the class declaration above for more information.
 41 	initialize: function()
 42 	{
 43 		//do nothing
 44 	},
 45 	
 46 	/**
 47 	 * The Base setUp function for the Test Harness. setUp() is run before every 
 48 	 * test case in a test set.
 49 	 * 
 50 	 * @returns {undefined} 
 51 	 * 					The framework expects no values to be returned 
 52 	 * 					from this function.
 53 	 */
 54 	setUp: function() 
 55 	{
 56 		//do nothing
 57 	},
 58 	
 59 	/**
 60 	 * The Base tearDown function for the Test Harness. tearDown() is run after 
 61 	 * every test case in a test set.
 62 	 * 
 63 	 * @returns {undefined} 
 64 	 * 					The framework expects no values to be returned 
 65 	 * 					from this function.
 66 	 */
 67 	tearDown: function()
 68 	{
 69 		//do nothing.
 70 	},
 71 	
 72 	/**
 73 	 * This is run before every test set. Note that it is a higher level setUp 
 74 	 * function than the plain setUp function because it sets up a whole set of 
 75 	 * test cases at once.
 76 	 * 
 77 	 * @returns {undefined} 
 78 	 * 					The framework expects no values to be returned 
 79 	 * 					from this function.
 80 	 */
 81 	setUpTestSet: function()
 82 	{
 83 		//do nothing.
 84 	},
 85 	
 86 	/**
 87 	 * This is run after every test set. Note that it is a higher level tearDown 
 88 	 * function than the plain tearDown function because it tears down a whole 
 89 	 * set of test cases at once.
 90 	 *
 91 	 * @returns {undefined} 
 92 	 * 					The framework expects no values to be returned 
 93 	 * 					from this function.
 94 	 */
 95 	tearDownTestSet: function()
 96 	{
 97 		//do nothing.
 98 	},
 99 	
100 	/**
101 	 * Do remaining cleanup work to clean up after test failures.
102 	 * 
103 	 * @returns {undefined} 
104 	 * 					The framework expects no values to be returned 
105 	 * 					from this function.
106 	 */
107 	doCleanup: function()
108 	{
109 		//do nothing.
110 	}
111 });
