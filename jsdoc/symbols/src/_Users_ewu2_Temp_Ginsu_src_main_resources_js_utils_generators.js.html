 11 /**
 12  * Generate a random number between minVal and maxVal.
 13  * @param {number} minVal 
 14  * 					The minimum allowed value in the range of randomly generated 
 15  * 					numbers.
 16  * @param {number} maxVal 
 17  * 					The maximum allowed value in the range of randomly generated 
 18  * 					numbers.
 19  * @param {boolean} floatVal 
 20  * 					true if the randomly generated number should be a fload 
 21  * 					value. If true, will return an integer.
 22  * 
 23  * @return {number} a random number between minVal and maxVal.
 24  */
 25 IMAT.randomXToY = function(minVal, maxVal, floatVal)
 26 {
 27   var randVal = minVal + (Math.random() * (maxVal - minVal));
 28   if(floatVal)
 29   {
 30   	randVal = Math.round(randVal);
 31   }
 32   else
 33   {
 34   	randVal = randVal.toFixed(floatVal);
 35   }
 36   return randVal;
 37 };

 39 /**
 40  * Get the current time in milliseconds.
 41  * @return {number} The number of milliseconds since Jan 1, 1970 (Unix Epoch) 
 42  */
 43 IMAT.getTime = function()
 44 {
 45 	var d = new Date();
 46 	return d.getTime();
 47 };

 49 /**
 50  * Generate a psuedo random phone number.
 51  * @return {string} A random phone number of the form NNN-NNN-NNNN 
 52  */
 53 IMAT.generatePhoneNumber = function()
 54 {
 55 	return IMAT.randomXToY(100,999) + "-" + IMAT.randomXToY(100,999) + "-" + IMAT.randomXToY(1000,9999) ;
 56 };
